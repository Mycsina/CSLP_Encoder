263void binarize(Image &im) {
264 Mat matrix = *im._get_image_mat();
265 int pix_num = matrix.rows * matrix.cols;
266 int bins = 256;
267 if (matrix.depth() != CV_8U) {
268 throw std::runtime_error(
269 "Original matrix must have 8-bit unsigned integers");
270 }
271 if (matrix.channels() == 3) {
272 cout << "Converting to grayscale" << endl;
273 auto gray = convert_BGR_GRAY(im);
274 matrix = *gray._get_image_mat();
275 }
276 Mat hist = histogram<uchar>(matrix, bins);
277 // Get empirical cumulative distribution function
278 Mat distrib = ecdf(hist, pix_num);
279 // Get best threshold
280 int best_threshold = 0;
281 double max_variance = 0;
282 for (int thresh = 1; thresh < bins - 1; thresh++) {
283 // Calculate percentage of pixels above and below threshold (foreground and
284 // background)
285
286 double fg_prob = distrib.at<float>(thresh);// w0
287 double bg_prob = 1 - fg_prob; // w1
288 // Skip if one of the percentages is 0
289 if (fg_prob == 0 || bg_prob == 0) {
290 continue;
291 }
292 double fg_mean = 0;// μ0
293 double bg_mean = 0;// μ1
294 for (int i = 0; i < thresh; i++) {
295 fg_mean += i * (double) hist.at<float>(i);
296 }
297 for (int i = thresh; i < bins; i++) {
298 bg_mean += i * (double) hist.at<float>(i);
299 }
300 fg_mean /= fg_prob;
301 bg_mean /= bg_prob;
302 double inter_variance =
303 fg_prob * bg_prob * ((fg_mean - bg_mean) * (fg_mean - bg_mean));
304 if (inter_variance > max_variance) {
305 max_variance = inter_variance;
306 best_threshold = thresh;
307 }
308 }
309 // Binarize image
310 matrix.forEach<uchar>(
311 [&best_threshold](uchar &pixel, const int *position) -> void {
312 if (pixel > best_threshold) {
313 pixel = 255;
314 } else {
315 pixel = 0;
316 }
317 });
318}
