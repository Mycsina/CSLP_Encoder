136Image Image::gaussian_blur(cv::Mat blur) {
137 Image im = this->clone();
138 Mat *m = im._get_image_mat();
139 int totalBlur = int(sum(blur)[0]);
140 int radiusR, radiusC;
141
142 // Check if the blur matrix has proper dimensions
143 if (blur.rows % 2 == 0 || blur.cols % 2 == 0) {
144 throw std::invalid_argument("blur matrix must have odd rows and columns");
145 }
146 radiusR = blur.rows / 2;
147 radiusC = blur.cols / 2;
148
149 // iterate every pixel to apply the filter
150 image_mat_.forEach<Vec3b>([&blur, radiusC, radiusR, m, this,
151 totalBlur](Vec3b &pixel, const int *position) {
152 int row = position[0];
153 int col = position[1];
154
155 Mat blur_temp = cut(blur, row, col); // cut the blur filter as needed
156 Mat temp = get_neighbors(radiusR, radiusC, row, col);// get the neighbors
157
158 temp = temp.mul(blur_temp);
159 cv::Scalar sum = cv::sum(temp);
160
161 // normalize
162 for (int a = 0; a < sum.rows; a++) {
163 pixel[a] = (int) (sum[a] / totalBlur);
164 }
165 m->at<Vec3b>(row, col) = pixel;

183Mat Image::cut(const Mat &m, int row, int col) const {
184 int centerRow = m.rows / 2;
185 int centerCol = m.cols / 2;
186 // check distance in image_mat from this pixel to the edge
187 int distUp = row - 0;
188 int distDown = (image_mat_.rows - 1) - row;
189 int distLeft = col - 0;
190 int distRight = (image_mat_.cols - 1) - col;
191
192 // check for need to cut
193 int rMin = max((centerRow - distUp), 0);
194 int rMax = min((centerRow + distDown), m.rows - 1);
195 int cMin = max((centerCol - distLeft), 0);
196 int cMax = min((centerCol + distRight), m.cols - 1);
197
198 return m(Range(rMin, rMax + 1), Range(cMin, cMax + 1));
199}
