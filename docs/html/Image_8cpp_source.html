107vector<Mat> Image::color_histograms(int bins, bool fill_hist, int width,
108 int height) {
109 if (loaded()) {
110 vector<Mat> histograms;
111 vector<Mat> channels;
112 // Array of colors for each channel
113 Scalar colors[] = {Scalar(255, 0, 0), Scalar(0, 255, 0), Scalar(0, 0, 255)};
114 split(image_mat_, channels);
115 // If image is grayscale, set color to white
116 if (channels.size() == 1) {
117 colors[0] = Scalar(255, 255, 255);
118 }
119 for (int i = 0; i < channels.size(); i++) {
120 Histogram hist = Histogram(bins);
121 hist.color = colors[i];
122 Mat backMat = histogram<uchar>(channels[i], bins);
123 int bin_w = cvRound((double) width / 256);
124 Vec3b backColor = (0, 0, 0);
125
126 Mat histImage(height, width, CV_8UC3, backColor);
127
128 normalize(backMat, backMat, 0, histImage.rows, NORM_MINMAX, -1, Mat());
129
130 for (int j = 1; j < 256; j++) {
131 line(histImage,
132 Point(bin_w * (j - 1), height - cvRound(backMat.at<float>(j - 1))),
133 Point(bin_w * (j), height - cvRound(backMat.at<float>(j))),
134 hist.color, 2, 8, 0);
135 }
136
137 if (fill_hist) {
138 for (int j = 0; j < histImage.cols; j++) {
139 Mat col = histImage.col(j);
140 for (int k = col.rows - 1; k >= 0; k--) {
141 // Fill column with color until a non-<backColor> pixel is found
142 if (col.at<Vec3b>(k) != backColor) {
143 break;
144 }
145 Vec3b color = {(uchar) hist.color[0], (uchar) hist.color[1],
146 (uchar) hist.color[2]};
147 col.at<Vec3b>(k) = color;
148 // If we reach end of column without finding a non-<backColor> pixel
149 // zero the column
150 if (k == 0) {
151 col = Mat::zeros(col.rows, 1, CV_8UC3);
152 }
153 }
154 }
155 }
156
157 histograms.push_back(histImage);
158 }
159 return histograms;
160 } else
161 throw std::runtime_error("Image hasn't been loaded");
162}

164Image Image::gaussian_blur(cv::Mat blur) {
165 Image im = this->clone();
166 Mat *m = im._get_image_mat();
167 int totalBlur = int(sum(blur)[0]);
168 int radiusR, radiusC;
169
170 // Check if the blur matrix has proper dimensions
171 if (blur.rows % 2 == 0 || blur.cols % 2 == 0) {
172 throw std::invalid_argument("blur matrix must have odd rows and columns");
173 }
174 radiusR = blur.rows / 2;
175 radiusC = blur.cols / 2;
176
177 // iterate every pixel to apply the filter
178 image_mat_.forEach<Vec3b>([&blur, radiusC, radiusR, m, this,
179 totalBlur](Vec3b &pixel, const int *position) {
180 int row = position[0];
181 int col = position[1];
182
183 Mat blur_temp = cut(blur, row, col); // cut the blur filter as needed
184 Mat temp = get_neighbors(radiusR, radiusC, row, col);// get the neighbors
185
186 temp = temp.mul(blur_temp);
187 cv::Scalar sum = cv::sum(temp);
188
189 // normalize
190 for (int a = 0; a < sum.rows; a++) {
191 pixel[a] = (int) (sum[a] / totalBlur);
192 }
193 m->at<Vec3b>(row, col) = pixel;
194 });
195 return im;
196}

212Mat Image::cut(const Mat &m, int row, int col) const {
213 int centerRow = m.rows / 2;
214 int centerCol = m.cols / 2;
215 // check distance in image_mat from this pixel to the edge
216 int distUp = row;
217 int distDown = (image_mat_.rows - 1) - row;
218 int distLeft = col;
219 int distRight = (image_mat_.cols - 1) - col;
220
221 // check for need to cut
222 int rMin = max((centerRow - distUp), 0);
223 int rMax = min((centerRow + distDown), m.rows - 1);
224 int cMin = max((centerCol - distLeft), 0);
225 int cMax = min((centerCol + distRight), m.cols - 1);
226
227 return m(Range(rMin, rMax + 1), Range(cMin, cMax + 1));
228}
