106vector<Mat> Image::color_histograms(const int bins, const bool fill_hist, const int width,
107 const int height) const {
108 if (loaded()) {
109 vector<Mat> histograms;
110 vector<Mat> channels;
111 // Array of colors for each channel
112 Scalar colors[] = {Scalar(255, 0, 0), Scalar(0, 255, 0), Scalar(0, 0, 255)};
113 split(image_mat_, channels);
114 // If image is grayscale, set color to white
115 if (channels.size() == 1) {
116 colors[0] = Scalar(255, 255, 255);
117 }
118 for (int i = 0; i < channels.size(); i++) {
119 Histogram hist = Histogram(bins);
120 hist.color = colors[i];
121 Mat backMat = histogram<uchar>(channels[i], bins);
122 const int bin_w = cvRound(static_cast<double>(width) / 256);
123 Vec3b backColor = (0, 0, 0);
124
125 Mat histImage(height, width, CV_8UC3, backColor);
126
127 normalize(backMat, backMat, 0, histImage.rows, NORM_MINMAX, -1, Mat());
128
129 for (int j = 1; j < 256; j++) {
130 line(histImage,
131 Point(bin_w * (j - 1), height - cvRound(backMat.at<float>(j - 1))),
132 Point(bin_w * (j), height - cvRound(backMat.at<float>(j))),
133 hist.color, 2, 8, 0);
134 }
135
136 if (fill_hist) {
137 for (int j = 0; j < histImage.cols; j++) {
138 Mat col = histImage.col(j);
139 for (int k = col.rows - 1; k >= 0; k--) {
140 // Fill column with color until a non-<backColor> pixel is found
141 if (col.at<Vec3b>(k) != backColor) {
142 break;
143 }
144 const Vec3b color = {static_cast<uchar>(hist.color[0]), static_cast<uchar>(hist.color[1]),
145 static_cast<uchar>(hist.color[2])};
146 col.at<Vec3b>(k) = color;
147 // If we reach end of column without finding a non-<backColor> pixel
148 // zero the column
149 if (k == 0) {
150 col = Mat::zeros(col.rows, 1, CV_8UC3);
151 }
152 }
153 }
154 }
155
156 histograms.push_back(histImage);
157 }
158 return histograms;
159 } else
160 throw std::runtime_error("Image hasn't been loaded");
161}

163Image Image::gaussian_blur(cv::Mat blur) {
164 Image im = this->clone();
165 Mat *m = im.get_image_mat();
166 int totalBlur = static_cast<int>(sum(blur)[0]);
167
168 // Check if the blur matrix has proper dimensions
169 if (blur.rows % 2 == 0 || blur.cols % 2 == 0) {
170 throw std::invalid_argument("blur matrix must have odd rows and columns");
171 }
172 int radiusR = blur.rows / 2;
173 int radiusC = blur.cols / 2;
174
175 // iterate every pixel to apply the filter
176 image_mat_.forEach<Vec3b>([&blur, radiusC, radiusR, m, this,
177 totalBlur](Vec3b &pixel, const int *position) {
178 const int row = position[0];
179 const int col = position[1];
180
181 const Mat blur_temp = cut(blur, row, col); // cut the blur filter as needed
182 Mat temp = get_neighbors(radiusR, radiusC, row, col);// get the neighbors
183
184 temp = temp.mul(blur_temp);
185 Scalar sum = cv::sum(temp);
186
187 // normalize
188 for (int a = 0; a < Scalar::rows; a++) {
189 pixel[a] = static_cast<int>(sum[a] / totalBlur);
190 }
191 m->at<Vec3b>(row, col) = pixel;
192 });
193 return im;
194}

210Mat Image::cut(const Mat &m, const int row, const int col) const {
211 const int center_row = m.rows / 2;
212 const int center_col = m.cols / 2;
213 // check distance in image_mat from this pixel to the edge
214 const int dist_up = row;
215 const int dist_down = (image_mat_.rows - 1) - row;
216 const int dist_left = col;
217 const int dist_right = (image_mat_.cols - 1) - col;
218
219 // check for need to cut
220 const int r_min = max(center_row - dist_up, 0);
221 const int r_max = min(center_row + dist_down, m.rows - 1);
222 const int c_min = max(center_col - dist_left, 0);
223 const int c_max = min(center_col + dist_right, m.cols - 1);
224
225 return m(Range(r_min, r_max + 1), Range(c_min, c_max + 1));
