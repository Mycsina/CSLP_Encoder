106vector<Mat> Image::color_histograms(int bins, bool fill_hist, int width,
107 int height) {
108 if (loaded()) {
109 vector<Mat> histograms;
110 vector<Mat> channels;
111 // Array of colors for each channel
112 Scalar colors[] = {Scalar(255, 0, 0), Scalar(0, 255, 0), Scalar(0, 0, 255)};
113 split(image_mat_, channels);
114 // If image is grayscale, set color to white
115 if (channels.size() == 1) {
116 colors[0] = Scalar(255, 255, 255);
117 }
118 for (int i = 0; i < channels.size(); i++) {
119 Histogram hist = Histogram(bins);
120 hist.color = colors[i];
121 Mat backMat = histogram<uchar>(channels[i], bins);
122 int bin_w = cvRound((double) width / 256);
123 Vec3b backColor = (0, 0, 0);
124
125 Mat histImage(height, width, CV_8UC3, backColor);
126
127 normalize(backMat, backMat, 0, histImage.rows, NORM_MINMAX, -1, Mat());
128
129 for (int j = 1; j < 256; j++) {
130 line(histImage,
131 Point(bin_w * (j - 1), height - cvRound(backMat.at<float>(j - 1))),
132 Point(bin_w * (j), height - cvRound(backMat.at<float>(j))),
133 hist.color, 2, 8, 0);
134 }
135
136 if (fill_hist) {
137 for (int j = 0; j < histImage.cols; j++) {
138 Mat col = histImage.col(j);
139 for (int k = col.rows - 1; k >= 0; k--) {
140 // Fill column with color until a non-<backColor> pixel is found
141 if (col.at<Vec3b>(k) != backColor) {
142 break;
143 }
144 Vec3b color = {(uchar) hist.color[0], (uchar) hist.color[1],
145 (uchar) hist.color[2]};
146 col.at<Vec3b>(k) = color;
147 // If we reach end of column without finding a non-<backColor> pixel
148 // zero the column
149 if (k == 0) {
150 col = Mat::zeros(col.rows, 1, CV_8UC3);
151 }
152 }
153 }
154 }
155
156 histograms.push_back(histImage);
157 }
158 return histograms;
159 } else
160 throw std::runtime_error("Image hasn't been loaded");
161}

163Image Image::gaussian_blur(cv::Mat blur) {
164 Image im = this->clone();
165 Mat *m = im.getImageMat();
166 int totalBlur = int(sum(blur)[0]);
167 int radiusR, radiusC;
168
169 // Check if the blur matrix has proper dimensions
170 if (blur.rows % 2 == 0 || blur.cols % 2 == 0) {
171 throw std::invalid_argument("blur matrix must have odd rows and columns");
172 }
173 radiusR = blur.rows / 2;
174 radiusC = blur.cols / 2;
175
176 // iterate every pixel to apply the filter
177 image_mat_.forEach<Vec3b>([&blur, radiusC, radiusR, m, this,
178 totalBlur](Vec3b &pixel, const int *position) {
179 int row = position[0];
180 int col = position[1];
181
182 Mat blur_temp = cut(blur, row, col); // cut the blur filter as needed
183 Mat temp = get_neighbors(radiusR, radiusC, row, col);// get the neighbors
184
185 temp = temp.mul(blur_temp);
186 cv::Scalar sum = cv::sum(temp);
187
188 // normalize
189 for (int a = 0; a < cv::Scalar::rows; a++) {
190 pixel[a] = (int) (sum[a] / totalBlur);
191 }
192 m->at<Vec3b>(row, col) = pixel;
193 });
194 return im;
195}

211Mat Image::cut(const Mat &m, int row, int col) const {
212 int centerRow = m.rows / 2;
213 int centerCol = m.cols / 2;
214 // check distance in image_mat from this pixel to the edge
215 int distUp = row;
216 int distDown = (image_mat_.rows - 1) - row;
217 int distLeft = col;
218 int distRight = (image_mat_.cols - 1) - col;
219
220 // check for need to cut
221 int rMin = max((centerRow - distUp), 0);
222 int rMax = min((centerRow + distDown), m.rows - 1);
223 int cMin = max((centerCol - distLeft), 0);
224 int cMax = min((centerCol + distRight), m.cols - 1);
225
226 return m(Range(rMin, rMax + 1), Range(cMin, cMax + 1));

323uchar Image::predict_JPEG_LS(Mat mat, int row, int col, int channel = 0) {
324 if (row < 0 || row >= mat.rows || col < 0 || col >= mat.cols) {
325 throw std::out_of_range("Pixel out of bounds");
326 }
327
328 uchar a, b, c;
329 if (mat.channels() > 1) {
330 if (row - 1 >= 0 && col >= 1) {
331 a = mat.at<Vec3b>(row, col - 1)[channel];
332 b = mat.at<Vec3b>(row - 1, col)[channel];
333 c = mat.at<Vec3b>(row - 1, col - 1)[channel];
334 } else if (row - 1 >= 0) {
335 a = 0;
336 b = mat.at<Vec3b>(row - 1, col)[channel];
337 c = 0;
338 } else if (col - 1 >= 0) {
339 a = mat.at<Vec3b>(row, col - 1)[channel];

343 a = 0;
344 b = 0;
