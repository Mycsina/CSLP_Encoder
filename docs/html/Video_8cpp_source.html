87void Video::readFrame(FILE *file, int width, int height, int uvWidth,
88 int uvHeight, CHROMA_SUBSAMPLING format) {
89 Image im;
90 char buffer[6];
91 im._set_color(YUV);
92 im._set_chroma(format);
93 Mat yPlane(height, width, CV_8UC1);
94 Mat uPlane(uvHeight, uvWidth, CV_8UC1);
95 Mat vPlane(uvHeight, uvWidth, CV_8UC1);
96 Mat frame(height, width, CV_8UC3);
97 vector<Mat> channels(3);
98
99 // remove the "FRAME", if exists
100 if (fread(buffer, sizeof(char), 6, file) != 6 && !feof(file)) {
101 throw runtime_error("incomplete reading");
102 } else if (string(buffer) != "FRAME\n") {
103 // we're already past the frame (or this frame doesn't specify that a frame has started
104 fseek(file, -6, SEEK_CUR);
105 std::cout << buffer << " went back" << std::endl;
106 }
107
108 if (feof(file)) {
109 return;
110 }
111
112 // read yPlane
113 if (fread(reinterpret_cast<char *>(yPlane.data), sizeof(uint8_t),
114 width * height, file) != width * height) {
115 throw runtime_error("yPlane reading not completed");
116 }
117
118 // read uPlane
119 if (fread(reinterpret_cast<char *>(uPlane.data), sizeof(uint8_t),
120 uvWidth * uvHeight, file) != uvWidth * uvHeight) {
121 throw runtime_error("uPlane reading not completed");
122 }
123
124 // read vPlane
125 if (fread(reinterpret_cast<char *>(vPlane.data), sizeof(uint8_t),
126 uvWidth * uvHeight, file) != uvWidth * uvHeight) {
127 throw runtime_error("vPlane reading not completed");
128 }
129
130 // resize u and v (if it's 4:4:4 they're already at the correct size)
131 if (format != YUV444) {
132 resize(uPlane, uPlane, Size(width, height));
133 resize(vPlane, vPlane, Size(width, height));
134 }
135
136 // merge the three channels
137 split(frame, channels);
138 channels[0] = yPlane;
139 channels[1] = uPlane;
140 channels[2] = vPlane;
141 merge(channels, frame);
142
143 im._set_image_mat(frame);
144
145 im_reel.push_back(im);
146}

148void Video::getHeaderData(FILE *file, int *width, int *height, float *fps) {
149 char header[90];
150 char discard[200];// should be enough for the rest
151 int frame_rate_num, frame_rate_den;
