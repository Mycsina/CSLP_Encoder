115void Video::read_frame(FILE *file, const int width, const int height, const int uvWidth,
116 const int uvHeight, const CHROMA_SUBSAMPLING format) {
117 Image im;
118 char buffer[6];
119 im.set_color(YUV);
120 im.set_chroma(format);
121 const Mat yPlane(height, width, CV_8UC1);
122 Mat uPlane(uvHeight, uvWidth, CV_8UC1);
123 Mat vPlane(uvHeight, uvWidth, CV_8UC1);
124 Mat frame(height, width, CV_8UC3);
125 vector<Mat> channels(3);
126
127 // remove the "FRAME", if exists
128 if (fread(buffer, sizeof(char), 6, file) != 6 && !feof(file)) {
129 throw runtime_error("incomplete reading");
130 }
131 if (string(buffer) != "FRAME\n") {
132 // we're already past the frame (or this frame doesn't specify that a frame has started
133 fseek(file, -6, SEEK_CUR);
134 std::cout << buffer << " went back" << std::endl;
135 }
136
137 if (feof(file)) {
138 return;
139 }
140
141 // read yPlane
142 if (fread(yPlane.data, sizeof(uint8_t),
143 width * height, file) != width * height) {
144 throw runtime_error("yPlane reading not completed");
145 }
146
147 // read uPlane
148 if (fread(uPlane.data, sizeof(uint8_t),
149 uvWidth * uvHeight, file) != uvWidth * uvHeight) {
150 throw runtime_error("uPlane reading not completed");
151 }
152
153 // read vPlane
154 if (fread(vPlane.data, sizeof(uint8_t),
155 uvWidth * uvHeight, file) != uvWidth * uvHeight) {
156 throw runtime_error("vPlane reading not completed");
157 }
158
159 // resize u and v (if it's 4:4:4 they're already at the correct size)
160 if (format != YUV444) {
161 resize(uPlane, uPlane, Size(width, height));
162 resize(vPlane, vPlane, Size(width, height));
163 }
164
165 // merge the three channels
166 split(frame, channels);
167 channels[0] = yPlane;
168 channels[1] = uPlane;
169 channels[2] = vPlane;
170 merge(channels, frame);
171
172 im.set_image_mat(frame);
173
174 im_reel.push_back(im);
175}
