265  Mat matrix = *im.get_image_mat();
266  int pix_num = matrix.rows * matrix.cols;
267  int bins = 256;
268  if (matrix.depth() != CV_8U) {
269  throw std::runtime_error(
270  "Original matrix must have 8-bit unsigned integers");
271  }
272  if (matrix.channels() == 3) {
273  cout << "Converting to grayscale" << endl;
274  auto gray = convert_BGR_GRAY(im);
275  matrix = *gray.get_image_mat();
276  }
277  Mat hist = histogram<uchar>(matrix, bins);
278  // Get empirical cumulative distribution function
279  Mat distrib = ecdf(hist, pix_num);
280  // Get best threshold
281  int best_threshold = 0;
282  double max_variance = 0;
283  for (int thresh = 1; thresh < bins - 1; thresh++) {
284  // Calculate percentage of pixels above and below threshold (foreground and
285  // background)
286 
287  double fg_prob = distrib.at<float>(thresh);// w0
288  double bg_prob = 1 - fg_prob; // w1
289  // Skip if one of the percentages is 0
290  if (fg_prob == 0 || bg_prob == 0) {
291  continue;
292  }
293  double fg_mean = 0;// μ0
294  double bg_mean = 0;// μ1
295  for (int i = 0; i < thresh; i++) {
296  fg_mean += i * static_cast<double>(hist.at<float>(i));
297  }
298  for (int i = thresh; i < bins; i++) {
299  bg_mean += i * static_cast<double>(hist.at<float>(i));
300  }
301  fg_mean /= fg_prob;
302  bg_mean /= bg_prob;
303  double inter_variance =
304  fg_prob * bg_prob * ((fg_mean - bg_mean) * (fg_mean - bg_mean));
305  if (inter_variance > max_variance) {
306  max_variance = inter_variance;
307  best_threshold = thresh;
308  }
309  }
310  // Binarize image
311  matrix.forEach<uchar>(
312  [&best_threshold](uchar &pixel, const int *) -> void {
313  if (pixel > best_threshold) {
314  pixel = 255;
315  } else {
316  pixel = 0;
317  }
318  });
319  im.set_image_mat(matrix);
320 }
