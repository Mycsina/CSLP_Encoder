8void watermark(Image &im, Image mark, Point2i coord1, Point2i coord2,
9 double alpha) {
10 Mat image_mat = *im.get_image_mat();
11 Mat markMat = *mark.get_image_mat();
12 int height = coord2.y - coord1.y;
13 int width = coord2.x - coord1.x;
14 // Resize the watermark to fit the desired area
15 InterpolationFlags flag;
16 if (height < markMat.rows && width < markMat.cols) {
17 flag = INTER_AREA;
18 } else {
19 flag = INTER_LINEAR;
20 }
21 resize(markMat, markMat, Size(width, height), 0, 0, flag);
22 // Blend the watermark with the Image
23 Mat roi = image_mat(Rect(coord1.x, coord1.y, width, height));
24 addWeighted(roi, alpha, markMat, 1.0 - alpha, 0.0, roi);
25}

205void subsample(Image &im, CHROMA_SUBSAMPLING cs) {
206 Mat *matrix = im.get_image_mat();
207 Mat channels[3];
208 split(*matrix, channels);
209 Size target_size = Size(channels[0].size[1], channels[0].size[0]);
210 float scaling[2];
211 if (matrix->cols % 2 != 0) {
212 throw std::runtime_error("Matrix must have an even number of columns");
213 }
214 switch (cs) {
215 case YUV444:
216 cout << "No subsampling has been performed" << endl;
217 break;
218 case YUV422:
219 // 1/2 horizontal sampling, full vertical sampling
220 scaling[0] = 0.5;
221 scaling[1] = 1;
222 break;
223 case YUV420:
224 // 1/2 horizontal sampling, 1/2 vertical sampling
225 scaling[0] = 0.5;
226 scaling[1] = 0.5;
227 break;
228 default:
229 throw std::runtime_error("Invalid subsampling ratio");
230 }
231 resize(channels[1], channels[1], target_size, scaling[0], scaling[1],
232 INTER_LINEAR);
233 resize(channels[2], channels[2], target_size, scaling[0], scaling[1],
234 INTER_LINEAR);
235 merge(channels, 3, *matrix);
236}

253Mat ecdf(const Mat &histogram, int total) {

255 float cumsum = 0;

264void binarize(Image &im) {
265 Mat matrix = *im.get_image_mat();
266 int pix_num = matrix.rows * matrix.cols;
267 int bins = 256;
268 if (matrix.depth() != CV_8U) {
269 throw std::runtime_error(
270 "Original matrix must have 8-bit unsigned integers");
271 }
272 if (matrix.channels() == 3) {
273 cout << "Converting to grayscale" << endl;
274 auto gray = convert_BGR_GRAY(im);
275 matrix = *gray.get_image_mat();
276 }
277 Mat hist = histogram<uchar>(matrix, bins);
278 // Get empirical cumulative distribution function
279 Mat distrib = ecdf(hist, pix_num);
280 // Get best threshold
281 int best_threshold = 0;
282 double max_variance = 0;
283 for (int thresh = 1; thresh < bins - 1; thresh++) {
284 // Calculate percentage of pixels above and below threshold (foreground and
285 // background)
286
287 double fg_prob = distrib.at<float>(thresh);// w0
288 double bg_prob = 1 - fg_prob; // w1
289 // Skip if one of the percentages is 0
290 if (fg_prob == 0 || bg_prob == 0) {
291 continue;
292 }
293 double fg_mean = 0;// μ0
294 double bg_mean = 0;// μ1
295 for (int i = 0; i < thresh; i++) {
296 fg_mean += i * static_cast<double>(hist.at<float>(i));
297 }
298 for (int i = thresh; i < bins; i++) {
299 bg_mean += i * static_cast<double>(hist.at<float>(i));
300 }
301 fg_mean /= fg_prob;
302 bg_mean /= bg_prob;
303 double inter_variance =
304 fg_prob * bg_prob * ((fg_mean - bg_mean) * (fg_mean - bg_mean));
305 if (inter_variance > max_variance) {
306 max_variance = inter_variance;
307 best_threshold = thresh;
308 }
309 }
310 // Binarize image
311 matrix.forEach<uchar>(
312 [&best_threshold](uchar &pixel, const int *) -> void {
313 if (pixel > best_threshold) {
314 pixel = 255;
315 } else {
316 pixel = 0;
317 }
318 });
319 im.set_image_mat(matrix);
320}
