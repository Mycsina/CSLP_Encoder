265uchar Frame::predict_JPEG_LS(Mat mat, int row, int col, int channel) {
266 if (row < 0 || row >= mat.rows || col < 0 || col >= mat.cols) {
267 throw std::out_of_range("Pixel out of bounds");
268 }
269
270 uchar a, b, c;
271 if (mat.channels() > 1) {
272 if (row - 1 >= 0 && col >= 1) {
273 a = mat.at<Vec3b>(row, col - 1)[channel];
274 b = mat.at<Vec3b>(row - 1, col)[channel];
275 c = mat.at<Vec3b>(row - 1, col - 1)[channel];
276 } else if (row - 1 >= 0) {
277 a = 0;
278 b = mat.at<Vec3b>(row - 1, col)[channel];
279 c = 0;
280 } else if (col - 1 >= 0) {
281 a = mat.at<Vec3b>(row, col - 1)[channel];
282 b = 0;
283 c = 0;
284 } else {
285 a = 0;
286 b = 0;
287 c = 0;
288 }
289 } else {
290 if (row - 1 >= 0 && col >= 1) {
291 a = mat.at<uchar>(row, col - 1);
292 b = mat.at<uchar>(row - 1, col);
293 c = mat.at<uchar>(row - 1, col - 1);
294 } else if (row - 1 >= 0) {
295 a = 0;
296 b = mat.at<uchar>(row - 1, col);
297 c = 0;
298 } else if (col - 1 >= 0) {
299 a = mat.at<uchar>(row, col - 1);
300 b = 0;
301 c = 0;
302 } else {
303 a = 0;
304 b = 0;
305 c = 0;
306 }
307 }
308
309
310 if (c >= std::max(a, b)) {
311 return std::min(a, b);
312 } else if (c <= std::min(a, b)) {
313 return std::max(a, b);
314 } else {
315 return a + b - c;
316 }

347 auto block_coords = block.getVertices();

349 double diff_value = block_diff(block, ref_block);
350 if (isBetter(diff_value)) {

352 mv.residual = block.getBlockMat() - ref_block.getBlockMat();
353 best_score = diff_value;
354 previous_best = best_match;
