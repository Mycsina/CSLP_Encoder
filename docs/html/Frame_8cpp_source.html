267uchar Frame::predict_JPEG_LS(Mat mat, const int row, const int col, const int channel) {
268 if (row < 0 || row >= mat.rows || col < 0 || col >= mat.cols) {
269 throw std::out_of_range("Pixel out of bounds");
270 }
271
272 uchar a, b, c;
273 if (mat.channels() > 1) {
274 if (row - 1 >= 0 && col >= 1) {
275 a = mat.at<Vec3b>(row, col - 1)[channel];
276 b = mat.at<Vec3b>(row - 1, col)[channel];
277 c = mat.at<Vec3b>(row - 1, col - 1)[channel];
278 } else if (row - 1 >= 0) {
279 a = 0;
280 b = mat.at<Vec3b>(row - 1, col)[channel];
281 c = 0;
282 } else if (col - 1 >= 0) {
283 a = mat.at<Vec3b>(row, col - 1)[channel];
284 b = 0;
285 c = 0;
286 } else {
287 a = 0;
288 b = 0;
289 c = 0;
290 }
291 } else {
292 if (row - 1 >= 0 && col >= 1) {
293 a = mat.at<uchar>(row, col - 1);
294 b = mat.at<uchar>(row - 1, col);
295 c = mat.at<uchar>(row - 1, col - 1);
296 } else if (row - 1 >= 0) {
297 a = 0;
298 b = mat.at<uchar>(row - 1, col);
299 c = 0;
300 } else if (col - 1 >= 0) {
301 a = mat.at<uchar>(row, col - 1);
302 b = 0;
303 c = 0;
304 } else {
305 a = 0;
306 b = 0;
307 c = 0;
308 }
309 }
310
311
312 if (c >= std::max(a, b)) {
313 return std::min(a, b);
314 }
315 if (c <= std::min(a, b)) {
316 return std::max(a, b);
317 }
318 return a + b - c;

343bool Block::BlockDiff::compare(const Block &block, const Frame *reference, const Point center) {
344 const auto block_coords = block.getVertices();
345 const Block ref_block = get_block(reference->get_image(), block.getSize(), center.y, center.x);
346 const double diff_value = block_diff(block, ref_block);
347 if (isBetter(diff_value)) {
348 MotionVector mv = {center.x - block_coords[0], center.y - block_coords[1]};
349 mv.residual = Mat::zeros(block.getBlockMat().size(), CV_16SC3);
350 for (int i = 0; i < mv.residual.rows; i++)
351 for (int j = 0; j < mv.residual.cols; j++) {
352 mv.residual.at<Vec3s>(i, j)[0] = block.getBlockMat().at<Vec3b>(i, j)[0] - ref_block.getBlockMat().at<Vec3b>(i, j)[0];
353 mv.residual.at<Vec3s>(i, j)[1] = block.getBlockMat().at<Vec3b>(i, j)[1] - ref_block.getBlockMat().at<Vec3b>(i, j)[1];
354 mv.residual.at<Vec3s>(i, j)[2] = block.getBlockMat().at<Vec3b>(i, j)[2] - ref_block.getBlockMat().at<Vec3b>(i, j)[2];
355 }
356 best_score = diff_value;
357 best_match = mv;
358 }
359 if (diff_value <= threshold)
360 return true;
361 return false;

402MotionVector Frame::match_block_arps(const Block &block, Frame *reference) const {
403 bool finished = false;
404 vector<Point> visited;

423#endif
424 if (find(visited.begin(), visited.end(), point) != visited.end())
