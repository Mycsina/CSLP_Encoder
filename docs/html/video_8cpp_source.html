69void video::readFrame(FILE *file, int width, int height, int uvWidth,
70 int uvHeight, CHROMA_SUBSAMPLING format) {
71 Image im;
72 char buffer[6];
73 im.setColor(YUV);
74 im.setChroma(format);
75 Mat yPlane(height, width, CV_8UC1);
76 Mat uPlane(uvHeight, uvWidth, CV_8UC1);
77 Mat vPlane(uvHeight, uvWidth, CV_8UC1);
78 Mat frame(height, width, CV_8UC3);
79 vector<Mat> channels(3);
80
81 // remove the "FRAME", if exists
82 if (fread(buffer, sizeof(char), 6, file) != 6 && !feof(file)) {
83 throw new std::runtime_error("incomplete reading");
84 } else if (string(buffer) !=
85 "FRAME\n") { // we're already past the frame (or this frame doesn't
86 // specify that a frame has started
87 fseek(file, -6, SEEK_CUR);
88 std::cout << buffer << " went back" << std::endl;
89 }
90
91 if (feof(file)) {
92 return;
93 }
94
95 // read yPlane
96 if (fread(reinterpret_cast<char *>(yPlane.data), sizeof(uint8_t),
97 width * height, file) != width * height) {
98 throw new runtime_error("yPlane reading not completed");
99 }
100
101 // read uPlane
102 if (fread(reinterpret_cast<char *>(uPlane.data), sizeof(uint8_t),
103 uvWidth * uvHeight, file) != uvWidth * uvHeight) {
104 throw new std::runtime_error("uPlane reading not completed");
105 }
106
107 // read vPlane
108 if (fread(reinterpret_cast<char *>(vPlane.data), sizeof(uint8_t),
109 uvWidth * uvHeight, file) != uvWidth * uvHeight) {
110 throw new std::runtime_error("vPlane reading not completed");
111 }
112
113 // resize u and v (if it's 4:4:4 they're already at the correct size)
114 if (format != YUV444) {
115 resize(uPlane, uPlane, Size(width, height));
116 resize(vPlane, vPlane, Size(width, height));
117 }
118
119 // merge the three channels
120 split(frame, channels);
121 channels[0] = yPlane;
122 channels[1] = uPlane;
123 channels[2] = vPlane;
124 merge(channels, frame);
125
126 im.setImageMat(frame);
127
128 im_reel.push_back(im);
129}
